To Create a DAX Measure, follow these steps:

Ã˜  Right Click and select New Measures and  Write the following DAX Measures

   

Total = COUNTA(LandSlide[event_id])
PerMth = CALCULATE([.Total],DATESYTD('Calendar'[Date],"12/31"))
YoY% =
VAR __PREV_YEAR =CALCULATE([.Total],
SAMEPERIODLASTYEAR('Calendar'[Date]))
RETURN
     DIVIDE([.Total] - __PREV_YEAR ,__PREV_YEAR,0)/7
MoM% =
VAR __PREV_MONTH =CALCULATE(COUNTA('LandSlide'[event_id]),
        DATEADD('Calendar'[Date], -1, MONTH))
RETURN
    DIVIDE(COUNTA('LandSlide'[event_id]) - __PREV_MONTH, __PREV_MONTH)
Countries =
CALCULATE([.Total],ALLEXCEPT(LandSlide,LandSlide[country_name]),values('Calendar'[Month]))

 [YoY%]>0,UNICHAR(9650),
[YoY%] <0,UNICHAR(9660))
.MOM% KPI = SWITCH(TRUE(),
 [MoM%]>0,UNICHAR(9650),
[MoM%] <0,UNICHAR(9660)
)
.MOM KPI Color = IF([.MOM% KPI]=UNICHAR(9660), "Green" , "Red")
